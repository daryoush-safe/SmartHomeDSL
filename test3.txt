device mainLED : LED pin 13;
device buzzer : BUZZER pin 8;
device tempSensor : TEMPERATURE_SENSOR pin 14;
device lightSensor : LIGHT_SENSOR pin 15;

state idle {
    mainLED.off()
}

state alert {
    mainLED.blink(500)
    buzzer.beep(1000)
}

state comfortable {
    mainLED.on()
}

    transition idle -> alert when tempSensor.getTemperature() * 3 > 15;
    transition idle -> comfortable when lightSensor.getLight() >= 100;
