device myServo : SERVO pin 9;
device humSensor : HUMIDITY_SENSOR pin 14;

state idle {
  myServo.moveServo(20)
}

state good {
  myServo.moveServo(120)
}

state new{
  myServo.sweepServo(46,120,1,20)
}

transition new -> idle when humSensor.getHumidity() < 30;
transition idle -> new when humSensor.getHumidity() >50;
